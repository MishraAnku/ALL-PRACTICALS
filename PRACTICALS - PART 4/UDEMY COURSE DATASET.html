# plot heatmap 

import pandas as pd
import seaborn as sns
import matplotlib.pyplot as plt

# Load the dataset
udemy_data = pd.read_csv("Udemy Courses.csv")

# Convert 'price' column to numeric, treating 'Free' as NaN
udemy_data['price'] = pd.to_numeric(udemy_data['price'], errors='coerce')

# Drop rows with NaN values in 'price' column
udemy_data = udemy_data.dropna(subset=['price'])

# Assuming you have columns suitable for a heatmap, you can select them accordingly
# For example, let's say you want to visualize the correlation between 'price' and 'num_subscribers'
heatmap_data = udemy_data[['price', 'num_subscribers']]

# Calculate the correlation matrix
correlation_matrix = heatmap_data.corr()

# Plotting the heatmap
plt.figure(figsize=(10, 8))
sns.heatmap(correlation_matrix, annot=True, cmap='coolwarm', fmt=".2f")
plt.title('Correlation Heatmap between Price and Number of Subscribers')
plt.show()

# # Find Total Number of Courses Related To Python.

import pandas as pd

# Load the Udemy courses dataset
df = pd.read_csv("Udemy Courses.csv")

# Check if 'course_description' column exists in the dataframe
if 'course_description' in df.columns:
    # Filter courses related to Python
    python_courses = df[df['course_title'].str.contains('Python', case=False) |
                        df['course_description'].str.contains('Python', case=False) |
                        df['headline'].str.contains('Python', case=False)]
    
    # Count the total number of Python courses
    total_python_courses = python_courses.shape[0]

    print("Total number of courses related to Python:", total_python_courses)
else:
    print("Error: 'course_description' column not found in the dataset.")
